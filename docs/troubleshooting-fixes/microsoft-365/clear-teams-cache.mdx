import Tabs from '@theme/Tabs';

<Admonition type="info" >
  <p>
   Be aware that clearing the cache will require your system to download all the data again. This may take a while depending on your internet connection speed.
  </p>
</Admonition>

<Tabs groupId="operating-systems">
  <TabItem value="mac" label="Mac" default>
    <ol>
    
    <li>If Teams is still running, right-click the Teams icon in the dock, and then select Quit, or press Command <kbd>⌘</kbd> + <kbd>Q</kbd>.</li>
    
    <li>In the Finder, open the <code>/Applications/Utilities</code> folder, and then <b>double-click</b> Terminal.</li>
    
    <li>Enter the following command, and then press Return: 
        <br/>
        <br/>
        <CodeBlock language="bash">rm -r ~/Library/Application\ Support/Microsoft/Team</CodeBlock>
    </li>
    <li>
        Navigate to Keychain Access.
        Search for ‘Microsoft Teams’ and delete the entry.
        Open Teams and wait for it to download required files.
    </li>
    <li>Restart Teams.</li>
    </ol>

  </TabItem>
  <TabItem value="windows" label="Windows">
    <ol>
        <li>If Teams is still running, right-click the Teams icon in the taskbar, and then select <b>Quit</b>.</li>
        <li>Open the <b>Run dialog box</b> by pressing the <kbd>Windows</kbd> logo key + <kbd>R</kbd>.</li>
        <li>In the <b>Run dialog box</b>, enter <code>%appdata%\Microsoft\Teams</code>, and then select <b>OK</b>.</li>
        <li><b>Delete</b> all files and folders in the <code>%appdata%\Microsoft\Teams</code> directory.</li>
        <li>Restart Teams.</li>
    </ol>

  </TabItem>
  </TabItem>
</Tabs>
